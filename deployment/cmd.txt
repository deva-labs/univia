[install k8s]
1. Remove Old Kubernetes Sources
sudo rm /etc/apt/sources.list.d/kubernetes.list
sudo apt update
2. Verify Snap Installation (or install via snap)
snap list
3Ô∏è. Enable Kubelet from Snap
sudo nano /etc/systemd/system/kubelet.service

[Unit]
Description=Kubernetes Kubelet
Documentation=https://kubernetes.io/docs/
After=network.target

[Service]
ExecStart=/snap/bin/kubelet
Restart=always
RestartSec=10
StartLimitInterval=0

[Install]
WantedBy=multi-user.target

4. Enable and Start Kubelet
sudo systemctl daemon-reload
sudo systemctl enable kubelet
sudo systemctl start kubelet

5.  Verify Kubelet Status
sudo systemctl status kubelet

6. Initialize Kubernetes Cluster Again
sudo kubeadm reset
sudo kubeadm init

7. Set Up Kubeconfig
mkdir -p $HOME/.kube
sudo cp -i /var/snap/kubeadm/common/etc/kubernetes/admin.conf $HOME/.kube/config
sudo chown $(id -u):$(id -g) $HOME/.kube/config

8. Check Cluster Status
kubectl cluster-info
kubectl get nodes
